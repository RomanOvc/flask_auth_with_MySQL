for MySQL: 
1) flask db init
2) flask db migrate

set FLASK_DEBUG=1
set FLASK_ENV=development

Step 3:
set MAIL_USERNAME=myusername@gmail.com
set MAIL_PASSWORD=youpassword

flask run
